-- AIRLINE DATA PROJECT

-- 1. CREATE TABLES

CREATE TABLE airlines (
    airline_id SERIAL PRIMARY KEY,
    airline_name TEXT UNIQUE
);

CREATE TABLE airports (
    airport_code TEXT PRIMARY KEY,
    airport_name TEXT,
    city TEXT,
    country TEXT
);

CREATE TABLE flights (
    flight_id SERIAL PRIMARY KEY,
    airline_id INT REFERENCES airlines(airline_id),
    origin_airport TEXT REFERENCES airports(airport_code),
    destination_airport TEXT REFERENCES airports(airport_code),
    year INT,
    month INT,
    day INT,
    departure_delay INT,
    arrival_delay INT
);

-- 2. BASIC CHECKS

SELECT * FROM airlines;
SELECT * FROM airports;
SELECT * FROM flights;

SELECT COUNT(*) FROM airlines;
SELECT COUNT(*) FROM airports;
SELECT COUNT(*) FROM flights;

-- 3. AIRLINES ANALYSIS

SELECT a.airline_name,
       COUNT(f.flight_id) AS total_flights
FROM airlines a
JOIN flights f ON a.airline_id = f.airline_id
GROUP BY a.airline_name
ORDER BY total_flights DESC;

SELECT a.airline_name,
       AVG(f.departure_delay) AS avg_departure_delay
FROM airlines a
JOIN flights f ON a.airline_id = f.airline_id
GROUP BY a.airline_name
ORDER BY avg_departure_delay DESC;

-- 4. AIRPORT ANALYSIS

SELECT ap.airport_code,
       ap.city,
       COUNT(f.flight_id) AS total_departures
FROM airports ap
JOIN flights f ON ap.airport_code = f.origin_airport
GROUP BY ap.airport_code, ap.city
ORDER BY total_departures DESC;

SELECT ap.airport_code,
       ap.city,
       COUNT(f.flight_id) AS total_arrivals
FROM airports ap
JOIN flights f ON ap.airport_code = f.destination_airport
GROUP BY ap.airport_code, ap.city
ORDER BY total_arrivals DESC;

-- 5. FLIGHT ANALYSIS
-- =========================

SELECT COUNT(*) AS total_flights FROM flights;

SELECT COUNT(*) AS delayed_flights
FROM flights
WHERE departure_delay > 15;

SELECT 
  ROUND(
    100.0 * SUM(CASE WHEN departure_delay <= 15 THEN 1 ELSE 0 END) / COUNT(*),
    2
  ) AS on_time_percentage
FROM flights;

SELECT month, COUNT(*) AS total_flights
FROM flights
GROUP BY month
ORDER BY month;

-- 6. FULL JOIN VIEW

SELECT 
    f.flight_id,
    a.airline_name,
    o.airport_code AS origin,
    d.airport_code AS destination,
    f.year,
    f.month,
    f.day,
    f.departure_delay,
    f.arrival_delay
FROM flights f
JOIN airlines a ON f.airline_id = a.airline_id
JOIN airports o ON f.origin_airport = o.airport_code
JOIN airports d ON f.destination_airport = d.airport_code;

-- 7. PERFORMANCE INSIGHTS
SELECT a.airline_name,
       AVG(f.departure_delay) AS avg_delay
FROM airlines a
JOIN flights f ON a.airline_id = f.airline_id
GROUP BY a.airline_name
ORDER BY avg_delay ASC
LIMIT 3;

SELECT a.airline_name,
       AVG(f.departure_delay) AS avg_delay
FROM airlines a
JOIN flights f ON a.airline_id = f.airline_id
GROUP BY a.airline_name
ORDER BY avg_delay DESC
LIMIT 3;
